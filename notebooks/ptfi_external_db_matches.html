02 - External Database Matches Visualization

Overview

This notebook visualizes the distribution of compounds matched to external chemical databases across different food categories in the PTFI dataset.

Background

After matching PTFI metabolomic features against external databases (DrugBank, FCCdb, and Agrochemical PubChem), we analyze how these database matches are distributed across food categories and identify the top food sources for compounds of interest.

Input Data

matched_data.csv - PTFI features with external database match flags

df_long.csv - PTFI long format data

Outputs

  1. Heatmap: Mean intensity of matched compounds across FoodOn Level 1 categories and databases
  2. Bar plots: Top 10 food sources for four selected bioactive compounds

Analysis Approach

We focus on compounds that matched to at least one external database and calculate mean intensities across food categories to identify patterns of distribution in food.

)

# Melt data to long format for databases
melted = df_matched_only.melt(
    id_vars=['foodon_level1_name', 'intensity'], 
    value_vars=['drugs', 'fcc', 'agro'],
    var_name='database_type', 
    value_name='match'
)

# Calculate mean intensity for matches only
pivot_df = melted[melted['match']].pivot_table(
    values='intensity', 
    index='foodon_level1_name', 
    columns='database_type', 
    aggfunc='mean'
)

# Rename columns to full database names
column_mapping = {
    'agro': 'Agrochemical PubChem',
    'drugs': 'DrugBank', 
    'fcc': 'FCCdb'
}
pivot_df.rename(columns=column_mapping, inplace=True)

# Sort by total abundance across databases
pivot_df['total'] = pivot_df.sum(axis=1)
pivot_df_sorted = pivot_df.sort_values('total', ascending=False).drop('total', axis=1)

# Reorder columns
column_order = ['DrugBank', 'FCCdb', 'Agrochemical PubChem']
pivot_df_final = pivot_df_sorted[column_order]


# Remove multi-component food product category
pivot_df_final = pivot_df_final[pivot_df_final.index != 'multi-component food product']

# Display the filtered data
print("Mean intensities by food category and database:")
pivot_df_final

# Define compounds of interest
compounds = {
    'Rotenone': 'Rotenone',
    'Phlorizin': 'Phlorizin',
    'Pulegone': '(+)-Pulegone|Pulegone',
    'Rosmarinic acid': 'Rosmarinic acid'
}

# Prepare data for each compound
compound_data = {}

for title, element_name in compounds.items():
    # Filter for compound
    data = df_matched_only[df_matched_only["element_name"] == element_name].copy()
    
    # Special filter for Rosmarinic acid (animal products only)
    if title == 'Rosmarinic acid':
        data = data[data["foodon_level1_name"] == "animal food product"]
    
    # Get top 10 food sources by mean intensity
    top_foods = (data.groupby("sample_foodon_leaf_name")["intensity"]
                     .mean()
                     .reset_index()
                     .sort_values("intensity", ascending=False)
                     .head(10))
    
    # Keep only non-zero intensities
    top_foods = top_foods[top_foods["intensity"] > 0]
    
    compound_data[title] = top_foods
    print(f"{title}: {len(top_foods)} food sources with non-zero intensity")

# Create 2x2 subplot grid
fig, ((ax1, ax2), (ax3, ax4)) = plt.subplots(2, 2, figsize=(6, 6), dpi=300)
plt.subplots_adjust(hspace=0.4, wspace=0.3)

axes = [ax1, ax2, ax3, ax4]
titles = list(compounds.keys())

# Plot each compound
for ax, title in zip(axes, titles):
    data = compound_data[title]
    
    # Sort by intensity (ascending for horizontal bars)
    sorted_data = data.sort_values('intensity', ascending=True)
    sorted_data = sorted_data.set_index('sample_foodon_leaf_name')
    
    # Plot horizontal bar chart
    sorted_data['intensity'].plot(
        kind='barh', 
        ax=ax, 
        color='skyblue', 
        alpha=0.7,
        width=sorted_data.shape[0] * 0.07
    )
    
    # Formatting
    ax.set_xlabel(title, fontsize=8)
    ax.set_ylabel('', fontsize=8)
    ax.tick_params(labelsize=7)
    ax.xaxis.set_major_locator(MaxNLocator(nbins=5, steps=[1, 2, 4, 6, 10]))
    ax.tick_params(axis='x', rotation=90)
    
    # Clean styling
    ax.spines['top'].set_visible(False)
    ax.spines['right'].set_visible(False)
    ax.grid(True, alpha=0.3, axis='x')

plt.tight_layout()
