03 - DrugBank therapeutic categories analysis

Overview

This notebook analyzes the distribution of pharmaceutical compounds across food categories based on their DrugBank therapeutic classifications.

Background

Compounds detected in PTFI that matched to DrugBank were classified by their therapeutic categories. This analysis reveals which pharmaceutical classes are present in different food types and examines the distribution of specific bioactive compounds with pharmaceutical relevance.

Input data

ptfi_data_long.csv - PTFI features in long format

  • sample_id, sample_foodon_leaf_name, metabolite, intensity

ptfi_drugs_categories.csv - DrugBank therapeutic categories

  • element_id: Metabolite identifier
  • categories: Pipe-separated therapeutic categories

ptfi_metadata.csv - Sample metadata

  • sample_id, foodon_level1_name

Outputs

  1. Heatmap: Mean intensity of therapeutic categories by food category
  2. Violin plots: Distribution of selected compounds across food types

def create_violinplot(df, compound_name, samples_to_mark, 
                      figsize=(7, 7), save_path=None, dpi=300):
    """
    Create a violin plot for one compound across food categories,
    marking specific samples in the animal category.
    
    Parameters:
    -----------
    df : DataFrame
        Data with columns: sample_foodon_leaf_name, foodon_level1_name, 
                          element_name, intensity
    compound_name : str
        Name of the compound to plot
    samples_to_mark : list
        List of sample names to highlight in animal products
    figsize : tuple
        Figure size (default (7, 7))
    save_path : str or None
        Path to save the figure
    dpi : int
        DPI for saved figure (default 300)
    """
    
    # Filter for the compound and intensity > 0
    compound_data = df[
        (df['element_name'] == compound_name) & 
        (df['intensity'] > 0)
    ].copy()
    
    # Remove NaN values
    compound_data = compound_data.dropna(subset=['foodon_level1_name'])
    
    if len(compound_data) == 0:
        print(f"Warning: No data found for {compound_name}")
        return None, None
    
    # Create figure
    fig, ax = plt.subplots(figsize=figsize)
    
    # Define colors for food categories
    category_colors = {
        'animal food product': '#FF6B6B',
        'plant food product': '#4ECDC4',
        'fungus food product': '#45B7D1',
        'algal food product': '#A78CEE'
    }
    
    # Get categories with data
    categories_with_data = compound_data['foodon_level1_name'].unique()
    
    # Create violin plot
    sns.violinplot(
        data=compound_data,
        x='foodon_level1_name',
        y='intensity',
        hue='foodon_level1_name',
        ax=ax,
        palette={k: v for k, v in category_colors.items() if k in categories_with_data},
        order=categories_with_data,
        legend=False
    )
    
    # Set y-axis to start at 0
    ax.set_ylim(bottom=0)
    
    # Define markers for marked samples
    markers = ['o', 's', '^', 'v', 'D', 'p', '*', 'X']
    
    # Filter for animal category and samples to mark
    animal_data = compound_data[
        (compound_data['foodon_level1_name'] == 'animal food product') &
        (compound_data['sample_foodon_leaf_name'].isin(samples_to_mark))
    ]
    
    # Get x position for animal category
    if 'animal food product' in categories_with_data:
        x_pos = list(categories_with_data).index('animal food product')
    else:
        x_pos = None
    
    # Legend setup
    legend_handles = []
    legend_labels = []
    
    # Plot marked samples
    if x_pos is not None:
        for i, sample_name in enumerate(samples_to_mark):
            marker = markers[i] if i < len(markers) else 'o'
            
            # Get data for this specific sample
            sample_data = animal_data[animal_data['sample_foodon_leaf_name'] == sample_name]
            
            if not sample_data.empty:
                # Plot the points
                ax.scatter(
                    [x_pos] * len(sample_data), 
                    sample_data['intensity'],
                    color='darkred',
                    s=100,
                    alpha=0.8,
                    zorder=5,
                    marker=marker,
                    edgecolors='white',
                    linewidth=1.5
                )
                
                # Add to legend
                handle = plt.Line2D([0], [0], marker=marker, color='w',
                                   markerfacecolor='darkred', markersize=12,
                                   markeredgecolor='white', markeredgewidth=1.5)
                legend_handles.append(handle)
                legend_labels.append(sample_name)
    
    # Add legend
    if legend_handles:
        ncol = min(len(legend_handles), 3)
        ax.legend(legend_handles, legend_labels,
                 loc='lower center', bbox_to_anchor=(0.5, 1.02),
                 fontsize=14, frameon=True, fancybox=False,
                 ncol=ncol, columnspacing=1.5, handletextpad=0.5)
    
    # Formatting
    ax.set_xlabel(compound_name, fontsize=16, fontweight='bold')
    ax.set_ylabel('Intensity', fontsize=16, fontweight='bold')
    ax.tick_params(axis='both', which='major', labelsize=14)
    plt.xticks(rotation=90, ha='right')
    
    plt.tight_layout()
    
    # Save if path provided
    if save_path is not None:
        import os
        save_dir = os.path.dirname(save_path)
        if save_dir and not os.path.exists(save_dir):
            os.makedirs(save_dir)
        fig.savefig(save_path, format='png', dpi=dpi, bbox_inches='tight')
        print(f"Plot saved to: {save_path}")
    
    return fig, ax
